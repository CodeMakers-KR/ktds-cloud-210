-- shopUsers1234 schema
CREATE TABLE USERS (
  USER_ID VARCHAR2(20 CHAR) NOT NULL,
  EMAIL VARCHAR2(100 CHAR) NOT NULL,
  PWD VARCHAR2(100 CHAR) NOT NULL,
  SALT VARCHAR2(100 CHAR) NOT NULL,
  NAME VARCHAR2(10 CHAR) NOT NULL,
  CREATED_AT DATE NOT NULL,
  CONSTRAINT USERS_PK PRIMARY KEY (USER_ID),
  CONSTRAINT USERS_UNIQUE UNIQUE (EMAIL)
);

CREATE SEQUENCE SHOP_USERS.SEQ_USERS_PK INCREMENT BY 1 MINVALUE 1 MAXVALUE 999999 CYCLE CACHE 10 NOORDER ;

-- shopItems1234 schema
CREATE TABLE ITEMS (
  ITEM_ID VARCHAR2(20 CHAR) NOT NULL,
  ITEM_NAME VARCHAR2(120 CHAR) NOT NULL,
  STOCK NUMBER(10,0) DEFAULT 0 NOT NULL,
  UNIT_PRICE NUMBER(10, 0) DEFAULT 0 NOT NULL,
  CREATE_AT DATE NOT NULL,
  CONSTRAINT ITEMS_PK PRIMARY KEY (ITEM_ID)
);

CREATE SEQUENCE SEQ_ITEMS_PK INCREMENT BY 1 MINVALUE 1 MAXVALUE 999999 CYCLE CACHE 10 NOORDER;


-- shopOrders1234 schema
CREATE TABLE ORDERS (
  ORDER_ID VARCHAR2(20 CHAR) NOT NULL,
  ITEM_ID VARCHAR2(20 CHAR) NOT NULL,
  ITEM_ORDER_COUNT NUMBER(5,0) NOT NULL,
  ITEM_UNIT_PRICE NUMBER(10,0) NOT NULL,
  ORDER_AMOUNT_PRICE NUMBER(10,0) NOT NULL,
  USER_ID VARCHAR2(20 CHAR) NOT NULL,
  CREATE_AT DATE NOT NULL,
  CONSTRAINT ORDERS_PK PRIMARY KEY (ORDER_ID)
);

CREATE SEQUENCE SEQ_ORDERS_PK INCREMENT BY 1 MINVALUE 1 MAXVALUE 999999 CYCLE CACHE 10 NOORDER ;
